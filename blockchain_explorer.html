// Data Generators function generateHash() { return '0x' + Array(64).fill(0).map(() => Math.floor(Math.random() * 16).toString(16)).join(''); } let currentBlock = 8942; // Load Real Data const realCerts = JSON.parse(localStorage.getItem('certichain_certificates') || '[]'); // Update Total Certs Stat // We'll add a baseline of ~14k + real certs const totalCertsDisplay = document.getElementById('total-certs'); if(totalCertsDisplay) { const baseline = 14203; totalCertsDisplay.innerText = (baseline + realCerts.length).toLocaleString(); } // Merge real certs into the "Latest Transactions" view // For visual effect, we'll treat the most recent certs as recent transactions function renderBlocks() { const container = document.getElementById('block-container'); let html = ''; for (let i = 0; i < 5; i++) { const blockNum=currentBlock - i; const txCount=Math.floor(Math.random() * 50) + 1; const time=i * 14 + Math.floor(Math.random() * 5); html +=`
BK

#${blockNum} ${i === 0 ? 'New' : ''}

${time} secs ago • ${txCount} txns

Miner: CertiChain Validator 01

${generateHash().substring(0, 20)}...
`; } container.innerHTML = html; } function renderTxs() { const container = document.getElementById('tx-container'); let html = ''; // Create a mixed list of real and mock transactions // Real certs are "Certificate Issue" let mixedTxs = []; // Add latest 3 real certs if they exist realCerts.slice(0, 3).forEach(cert => { mixedTxs.push({ hash: cert.certificateId || generateHash(), time: 'Recently', type: 'Certificate Issue', isMine: true }); }); // Fill the rest with mock data up to 8 while(mixedTxs.length < 8) { mixedTxs.push({ hash: generateHash(), time: Math.floor(Math.random() * 10) + ' mins ago' , type: Math.random()> 0.3 ? 'Certificate Issue' : 'Verification', isMine: false }); } // Sort? nicely, keep "Mine" at top for visibility // mixedTxs.sort(...) mixedTxs.forEach(tx => { const style = tx.isMine ? 'border-left: 3px solid #10b981; background: rgba(16, 185, 129, 0.1);' : ''; const icon = tx.isMine ? '🎓' : '📄'; const hashDisplay = tx.hash.length > 20 ? tx.hash.substring(0, 16) + '...' : tx.hash; html += `
${icon}
${hashDisplay}
${tx.time}
${tx.type} ${tx.isMine ? '(Yours)' : ''}
`; }); container.innerHTML = html; } // Initialize and Update renderBlocks(); renderTxs(); // Simulate Live Updates setInterval(() => { currentBlock++; document.getElementById('latest-block-num').innerText = '#' + currentBlock; renderBlocks(); // Randomly update stats if (Math.random() > 0.7) { const totalEl = document.getElementById('total-certs'); const total = parseInt(totalEl.innerText.replace(/,/g, '')); totalEl.innerText = (total + 1).toLocaleString(); } }, 5000);
